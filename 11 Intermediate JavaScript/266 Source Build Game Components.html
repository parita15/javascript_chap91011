        function shuffleArray(array) {
            for (var x = array.length - 1; x > 0; x--) {
                var holder = Math.floor(Math.random() * (x + 1));
                var itemValue = array[x];
                array[x] = array[holder];
                array[holder] = itemValue;
            }
            return array;
        }
