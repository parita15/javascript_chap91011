        function lookUp() {

            var st = document.getElementById('sText').value;
            var rt = document.getElementById('rText').value;
            var re = new RegExp(st, "gi");
            var result = s.match(re);
            if (result) {
                var rr = s.replace(re, rt);
                document.getElementById('output').innerText = rr;
            }
            console.log(result);
        }
